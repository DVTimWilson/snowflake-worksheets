-- Add double quotes to protect funky column names

USE DATABASE SHARED_SANDBOX
;

USE SCHEMA OSC_DATA_EXTRACTS
;

SELECT query_to_run || ' ORDER BY TableName, Pos;'
FROM (
    SELECT LISTAGG(TableQuery, ' UNION ALL ') AS query_to_run
    FROM (
        SELECT '
    SELECT
      TableName
      ,TotalRows
      ,SUBSTRING(Col, 1, CHARINDEX(''__'', Col) - 1) AS Col
      ,CAST(SUBSTRING(Col, CHARINDEX(''__'', Col) + 2) as INTEGER) AS Pos
      ,NullCount
      ,CAST(CAST(NullCount AS DECIMAL(12,2)) / NULLIF(CAST(TotalRows AS DECIMAL(12,2)),0) * 100.00 AS DECIMAL(12,2)) AS NullPct
    FROM (
        SELECT
          ''' || t.TABLE_NAME || ''' AS TableName,
          COUNT(*) AS TotalRows,
          ' || LISTAGG('COUNT(*) - COUNT("' || c.COLUMN_NAME || '") AS "' || c.COLUMN_NAME || '__' || c.ORDINAL_POSITION, '",') || '"
        FROM ' || t.TABLE_SCHEMA || '.' || t.TABLE_NAME || '
        ) AS t
    UNPIVOT (NullCount FOR Col IN 
        (' || LISTAGG('"' || CAST(c.COLUMN_NAME AS varchar()) || '__' || CAST(c.ORDINAL_POSITION AS varchar()), '",') || '")
    ) AS p'
    
            FROM SHARED_SANDBOX.INFORMATION_SCHEMA.TABLES t
            INNER JOIN INFORMATION_SCHEMA.COLUMNS c ON c.TABLE_NAME = t.TABLE_NAME AND c.TABLE_SCHEMA = t.TABLE_SCHEMA
            WHERE t.TABLE_SCHEMA = 'OSC_DATA_EXTRACTS'
                --AND t.TABLE_NAME = 'SV_OSC_P_T_ACT_OBJ' OR t.TABLE_NAME = 'SV_OSC_P_T_ACT_ASS'
            GROUP BY t.TABLE_NAME, t.TABLE_SCHEMA
    
        ) AS t(TableQuery)
) AS generate
;
